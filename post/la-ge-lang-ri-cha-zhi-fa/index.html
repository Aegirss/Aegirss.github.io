拉格朗日插值

给出nnn个点Pi(xi,yi)P_i(x_i,y_i)Pi​(xi​,yi​)，必定存在过这nnn个点的最多n−1n-1n−1次的多项式f(x)f(x)f(x)，求f(k)f(k)f(k)的值。

原理：

对每个点做一个对于xxx轴的垂线，垂足记作Hi(xi,0)H_i(x_i,0)Hi​(xi​,0)。

每次选择一个点PiP_iPi​，然后设gi(x)g_i(x)gi​(x)是过PiP_iPi​与Hj(j≠i)H_j(j ≠ i)Hj​(j​=i)这nnn个点的最多n−1n-1n−1次的多项式。因为点HjH_jHj​都在xxx轴上，所以容易得到：

gi(x)=yi∗∏j≠ix−xjxi−xjg_i(x) = y_i * \prod_{j≠i} \frac{x-x_j}{x_i-x_j} gi​(x)=yi​∗j​=i∏​xi​−xj​x−xj​​

因为对于每个点PiP_iPi​都只有一个多项式gi(x)g_i(x)gi​(x)经过它，于是将所有的gi(x)g_i(x)gi​(x)相加，得到的多项式必定过这nnn个点，即：

f(x)=∑i=1ngi(x)=∑i=1n(yi∗∏j≠ix−xjxi−xj)f(x)=\sum_{i=1}^{n}g_i(x) = \sum_{i=1}^{n}(y_i * \prod_{j≠i} \frac{x-x_j}{x_i-x_j}) f(x)=i=1∑n​gi​(x)=i=1∑n​(yi​∗j​=i∏​xi​−xj​x−xj​​)

要求f(k)f(k)f(k)的值，只需把式子中的xxx变成kkk计算即可。计算的时候可以先分别计算分子和分母，再计算逆元，时间复杂度为O(n2)O(n^2)O(n2)。

代码实现：

#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll MOD = 998244353;
const int MAXN = 2010;

ll Pow(ll a, ll b)
{
	ll ans = 1;
	for(; b; b >>= 1)
	{
		if(b & 1) ans = ans * a % MOD;
		a = a * a % MOD;
	}
	return ans;
}

int n; ll ans, k;
ll x[MAXN], y[MAXN];

int main()
{
	scanf("%d %lld", &n, &k);
	for(int i = 1; i <= n; ++i) scanf("%lld %lld", &x[i], &y[i]);
	for(int i = 1; i <= n; ++i)
	{
		ll a = 1, b = 1;
		for(int j = 1; j <= n; ++j)
		{
			if(j == i) continue;
			a = a * (k - x[j]) % MOD;
			b = b * (x[i] - x[j]) % MOD;
		}
		a = (a + MOD) % MOD, b = (b + MOD) % MOD;
		ans = (ans + y[i] * a % MOD * Pow(b, MOD - 2) % MOD) % MOD;
	}
	printf("%lld\n", ans);
	return 0;
}
