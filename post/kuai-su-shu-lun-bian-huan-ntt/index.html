快速数论变换（NTT）

发布于 2020-04-21 | 9分钟 | 1602字数

简介：

快速傅里叶变换（FFT）运用了单位复根的性质来减少运算量，但是复数的实部和虚部都需要用浮点数来表示，大量的浮点数运算不仅慢，而且影响了精度，因此有了以数论为基础的具有循环卷积性质的快速数论变换(NTT)。

在FFT中，通过nnn次单位复根即ωn\omega_nωn​来运算，而对于NTT来说，则是运用了素数的原根来运算。

NTT：

PS：根据上面的性质，显然PPP必须是质数且N ∣ P−1N \ | \ P-1N ∣ P−1，而NNN又是222的幂，所以PPP应该是形如c∗2k+1c*2^k+1c∗2k+1的质数，如：998244353=119∗223+1998244353=119*2^{23}+1998244353=119∗223+1。

实现代码如下（多项式乘法）：

#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const ll MOD = 998244353;
const ll G = 3, Gi = 332748118;
const int MAXN = 4000010;

ll Pow(ll a, ll b)
{
	ll ans = 1;
	for(; b; b >>= 1)
	{
		if(b & 1) ans = ans * a % MOD;
		a = a * a % MOD;
	}
	return ans;
}

int rev[MAXN];
void change(ll y[], int len)
{
	for(int i = 0; i < len; ++i)
	{
		rev[i] = rev[i >> 1] >> 1;
		if(i & 1) rev[i] |= len >> 1;
	}
	for(int i = 0; i < len; ++i)
		if(i < rev[i]) swap(y[i], y[rev[i]]);
}

void NTT(ll y[], int len, int op)
{
	change(y, len);
	for(int h = 2; h <= len; h <<= 1)
	{
		ll wn = Pow((op == 1 ? G : Gi), (MOD - 1) / h);
		for(int j = 0; j < len; j += h)
		{
			ll step = 1;
			for(int k = j; k < j + h / 2; ++k)
			{
				ll u = y[k], t = step * y[k + h / 2] % MOD;
				y[k] = (u + t) % MOD;
				y[k + h / 2] = (u - t + MOD) % MOD;
				step = step * wn % MOD;
			}
		}
	}
	if(op == -1)
	{
		ll inv = Pow(len, MOD- 2);
		for(int i = 0; i < len; ++i) y[i] = y[i] * inv % MOD;
	}
}

ll f1[MAXN], f2[MAXN];

int main()
{
	int len1, len2, len = 1;
	scanf("%d %d", &len1, &len2);
	for(int i = 0; i <= len1; ++i) scanf("%lld", &f1[i]);
	for(int i = 0; i <= len2; ++i) scanf("%lld", &f2[i]);
	while(len <= len1 + len2) len <<= 1;
	NTT(f1, len, 1);
	NTT(f2, len, 1);
	for(int i = 0; i < len; ++i) f1[i] = f1[i] * f2[i] % MOD;
	NTT(f1, len, -1);
	for(int i = 0; i <= len1 + len2; ++i) printf("%lld ", f1[i]);
	putchar('\n');
	return 0;
}
