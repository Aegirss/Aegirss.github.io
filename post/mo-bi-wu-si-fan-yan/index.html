整除分块：

好吧这就是一个前置小技巧，来看这样一个式子：

∑i=1n⌊ni⌋\sum_{i=1}^{n} \lfloor \frac{n}{i} \rfloor i=1∑n​⌊in​⌋

利用整除分块的方法，我们可以在O((n))O(\sqrt(n))O((​n))的时间内计算出这个式子
容易发现，有许多⌊ni⌋\lfloor \frac{n}{i} \rfloor⌊in​⌋的值是一样的，而且它们都呈块状分布，经过一些小计算后可以发现，每一块的右边界即为n/(n/l)n/(n/l)n/(n/l)，于是我们就可以快速计算了

for(int l = 1, r; l <= n; l = r + 1)
{
    r = n / (n / l);
    ans += (r - l + 1) * (n / l);
}

// luogu-judger-enable-o2
#include<bits/stdc++.h>
using namespace std;

const int MAXN = 10000010;

int cnt, prime[MAXN], mu[MAXN], g[MAXN];
bool vis[MAXN]; long long sum[MAXN];

void get_mu(int n)
{
    mu[1] = 1;
    for(int i = 2; i <= n; ++i)
    {
        if(!vis[i]) {prime[++cnt] = i; mu[i] = -1;}
        for(int j = 1; j <= cnt && i * prime[j] <= n; ++j)
        {
            vis[i * prime[j]] = 1;
            if(i % prime[j] == 0) break;
            mu[i * prime[j]] = -mu[i];
        }
    }
    for(int j = 1; j <= cnt; ++j)
        for(int i = 1; i * prime[j] <= n; ++i)
            g[i * prime[j]] += mu[i];
    for(int i = 1; i <= n; ++i) sum[i] = sum[i - 1] + g[i];
}

int T, n, m;
long long ans;

int main()
{
    get_mu(MAXN - 10);
    scanf("%d", &T);
    while(T--)
    {
        scanf("%d %d", &n, &m);
        if(n > m) swap(n, m); ans = 0;
        for(int l = 1, r; l <= n; l = r + 1)
        {
            r = min(n / (n / l), m / (m / l));
            ans += (long long)(n / l) * (m / l) * (sum[r] - sum[l - 1]);
        }
        printf("%lld\n", ans);
    }
    return 0;
}

const int MAXN = 50010;

int cnt, prime[MAXN], mu[MAXN], sum[MAXN];

void init(int n)
{
    mu[1] = 1;
    for(int i = 2; i <= n; ++i)
    {
        if(!vis[i]) {prime[++cnt] = i; mu[i] = -1;}
        for(int j = 1; j <= cnt && prime[j] * i <= n; ++j)
        {
            vis[i * prime[j]] = 1;
            if(i % prime[j] == 0) break;

 #include<bits/stdc++.h>
using namespace std;

const int MAXN = 50010;
typedef long long ll;

int mu[MAXN], cnt, prime[MAXN];
bool vis[MAXN];

void get_mu(int n)
{
    mu[1] = 1;
    for(int i = 2; i <= n; ++i)
    {
        if(!vis[i]) prime[++cnt] = i, mu[i] = -1;
        for(int j = 1; j <= cnt && i * prime[j] <= n; ++j)
        {
            vis[i * prime[j]] = 1;
            if(i % prime[j] == 0) break;
            mu[i * prime[j]] = -mu[i];
        }
    }
    for(int i = 1; i <= n; ++i) mu[i] += mu[i - 1];
}

ll Calc(int n, int m, int k)
{
    ll ans = 0; if(n > m) swap(n, m);

    {
